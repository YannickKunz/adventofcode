# adventofcode

# UV Cheat Sheet

uv init                   # create new project
uv run x.py               # run a python program
uv add NAME               # add package
uv remove NAME            # remove package
uv add --dev NAME         # add dev dependency
uv sync                   # install all deps
uv run CMD                # run command inside env
uv lock --upgrade         # upgrade all packages
uv venv destroy           # delete virtual env
